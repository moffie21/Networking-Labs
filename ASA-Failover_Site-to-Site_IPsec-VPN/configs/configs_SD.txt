SD_ASA (5506) (L3) device
====================================
enable password P@ssw0rd
!
hostname SD_ASA
domain-name CIS.EDU
!
interface g0/0
 nameif outside
 security-level 0
 ip address 3.3.20.66 255.255.255.252
 no shut
!
interface g0/1
 nameif inside
 security-level 100
 ip address 10.20.128.1 255.255.255.252
 no shut
!
route outside 0.0.0.0 0.0.0.0 3.3.20.65
!
!
! Allowing ICMP through ASA
!
policy-map global_policy
 class inspection_default
  inspect icmp 
  inspect icmp error
!
!
! Allowing ICMP to ASA's inside interface from another site.
!
management-access inside
!
!
! Phase 1 (IKEv1)
!
crypto ikev1 enable outside
!
crypto ikev1 policy 10
 hash sha
 authentication pre-share
 group 5
 lifetime 86400
 encryption aes-256
!
tunnel-group 2.2.10.2 type ipsec-l2l
tunnel-group 2.2.10.2 ipsec-attributes
 ikev1 pre-shared-key LA10SD20
!
tunnel-group 4.4.21.130 type ipsec-l2l
tunnel-group 4.4.21.130 ipsec-attributes
 ikev1 pre-shared-key SD20SF21
!
!
! Phase 2 (IPSec)
!
object network N_10.10.0.0_16
 subnet 10.10.0.0 255.255.0.0
object network N_10.20.0.0_16
 subnet 10.20.0.0 255.255.0.0
object network N_10.21.0.0_16
 subnet 10.21.0.0 255.255.0.0
!
access-list IPSEC_LA_ACL extended permit ip object N_10.20.0.0_16 object N_10.10.0.0_16
!
access-list IPSEC_SF_ACL extended permit ip object N_10.20.0.0_16 object N_10.21.0.0_16
!
!
! Auto NAT
!
object network INSIDE_SD
 subnet 10.20.0.0 255.255.0.0
 nat (inside,outside) dynamic interface
!
!
! NAT Exemption (No NAT)
!
nat (inside,outside) source static N_10.20.0.0_16 N_10.20.0.0_16 destination static N_10.10.0.0_16 N_10.10.0.0_16 no-proxy-arp route-lookup
nat (inside,outside) source static N_10.20.0.0_16 N_10.20.0.0_16 destination static N_10.21.0.0_16 N_10.21.0.0_16 no-proxy-arp route-lookup
!
!
crypto ipsec ikev1 transform-set IPSEC_SET esp-aes-256 esp-sha-hmac
!
crypto map IPSEC_MAP 10 match address IPSEC_LA_ACL
crypto map IPSEC_MAP 10 set peer 2.2.10.2
crypto map IPSEC_MAP 10 set ikev1 transform-set IPSEC_SET
crypto map IPSEC_MAP 10 set security-association lifetime seconds 86400
crypto map IPSEC_MAP 10 set pfs group14
!
crypto map IPSEC_MAP 20 match address IPSEC_SF_ACL
crypto map IPSEC_MAP 20 set peer 4.4.21.130
crypto map IPSEC_MAP 20 set ikev1 transform-set IPSEC_SET
crypto map IPSEC_MAP 20 set security-association lifetime seconds 86400
crypto map IPSEC_MAP 20 set pfs group14
!
crypto map IPSEC_MAP interface outside
!
!
! OSPF
!
router ospf 10
 network 10.20.128.0 255.255.255.252 area 0
!

SD_3650 (L3/L2) (VTP Server) (Inter-VLAN Routing) device
====================================
hostname SD_3650
ip routing
!
interface g0/1
 no switchport
 ip address 10.20.128.2 255.255.255.252
 no shutdown
!
ip route 0.0.0.0 0.0.0.0 10.20.128.1
!
!
! VLANs
!
vlan 10
 name Management
vlan 40
 name Users
vlan 50
 name Wireless
!
!
! VTP
!
vtp mode server
vtp domain SD
vtp password cisco
!
!
! Inter-VLAN Routing
!
interface vlan 10
 description Mgmt VLAN
 ip addr 10.20.32.1 255.255.255.128
 no shutdown
!
interface vlan 40
 description Users VLAN
 ip addr 10.20.160.1 255.255.240.0
 no shutdown
!
interface vlan 50
 description Wireless VLAN
 ip addr 10.20.224.1 255.255.224.0
 no shutdown
!
interface g1/0
 switchport trunk encapsulation dot1q
 switchport mode trunk
 switchport trunk allowed vlan 10,40,50
 no shutdown
!
!
! OSPF
!
router ospf 10
 network 10.20.0.0 0.0.255.255 area 0
 passive-interface vlan 10
 passive-interface vlan 40
 passive-interface vlan 50
!
