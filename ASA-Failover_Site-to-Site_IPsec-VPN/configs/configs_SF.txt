FW_SF_R (2911) (Inter-VLAN Routing) device
====================================
hostname FW_SF_R
!
interface g0/0
 ip address 4.4.21.130 255.255.255.252
 ip nat outside
 no shutdown
!
interface g0/1
 no shutdown
!  
interface g0/1.10
 encapsulation dot1q 10
 ip address 10.21.32.1 255.255.255.128
 description ## Mgmt VLAN ##
 ip nat inside
 no shutdown
!
interface g0/1.40
 encapsulation dot1q 40
 ip address 10.21.160.1 255.255.240.0
 description ## Users VLAN ##
 ip nat inside
 no shutdown
!
interface g0/1.50
 encapsulation dot1q 50
 ip address 10.21.224.1 255.255.224.0
 description ## Wireless VLAN ##
 ip nat inside
 no shutdown
!
ip route 0.0.0.0 0.0.0.0 4.4.21.129
!
!
! Phase 1 (IKEv1)
!
crypto isakmp enable
!
crypto isakmp policy 10
 hash sha
 authentication pre-share
 group 5
 lifetime 86400 
 encryption aes 256
!
crypto isakmp key SD20SF21 address 3.3.20.66
!
!
! Phase 2 (IPsec)
!
ip access-list extended IPSEC_SD_LIST
 permit ip 10.21.0.0 0.0.255.255 10.20.0.0 0.0.255.255
!
!
! NAT (PAT)
!
ip access-list extended NAT_LIST
 deny ip 10.21.0.0 0.0.255.255 10.20.0.0 0.0.255.255
 permit ip 10.21.0.0 0.0.255.255 any
!
ip nat inside source list NAT_LIST interface g0/0 overload
!
!
crypto ipsec transform-set IPSEC_SET esp-aes 256 esp-sha-hmac
!
crypto map IPSEC_MAP 10 ipsec-isakmp
 match address IPSEC_SD_LIST
 set peer 3.3.20.66
 set transform-set IPSEC_SET
 set pfs group14
 set security-association lifetime seconds 86400
!
int g0/0
 crypto map IPSEC_MAP
!
